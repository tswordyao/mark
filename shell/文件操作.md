```
du -h *

# 只看一级目录 linux
du -h --max-depth=1

# mac上面的命令不行? 用这个:
du  -depth=1       这样输出该目录
du  -depth=1 *     这样输出一级目录
du  -h *           这样输出以M为单位
du  -h -depth=1 *  这样输出以M为单位并只有一级目录
du  -h -depth=1 * | sort -r  按大小倒序,很蠢, 是按字符顺序, 不是数字顺序
du  -depth=1 * | sort -r -n  全部按字节,再桉数字, 比如4.0M和5.0k
du  -m -depth=1 * | sort -r -n  全部按MB, 再按数字大小排序

# 排序并输出到文件
du -s * | sort -rn > file-size.txt 
```

```
echo:输出字符串或变量值
>   指定的文件若不存在，创建文件；若存在，覆盖原文件内容
输出重定向符

>>  指定的文件若不存在，创建文件；若存在，在原文件内容后追加内容

创建文件 vi filename
或vim touch cp mv tail cat less more ls echo


区别
  echo “字符串” > file
  echo “字符串” >> file
```
